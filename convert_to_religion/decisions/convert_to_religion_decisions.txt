	convert_to_hellenic_decision = {
		major = yes
		potential = {
			has_country_flag = ctr_religious_debate

					religion = hellenic_pagan
				}
			}
				# owns = 118 # Roma
				# owns = 146 # Athens
				# owns = 358 # Alexandria
				# owns = 148 # Thessaloniki
				# owns = 2296 # Biga
			NOT = { religion = hellenic_pagan }

		allow =  {
			is_at_war = no
			is_free_or_tributary_trigger = yes
			stability = 1
			OR = {
				AND = { # If any of these countries or culture, only need to hold 3 Holy Sites
					calc_true_if = {
						tag = ROM # Roman Empire
						tag = BYZ # Byzantine Empire
						tag = GRE # Greece
						culture_group = byzantine
						culture_group = latin
						amount = 1
					}
					calc_true_if = {
						all_owned_province = {
							OR = {
								province_id = 118
								province_id = 146
								province_id = 358
								province_id = 148
								province_id = 2296
							}
							is_core = ROOT
							OR = {
								has_building = temple
								has_building = cathedral
							}
						}
						amount = 3
					}
				}
				AND = { # Else you need all 5 Holy Sites
					calc_true_if = {
						all_owned_province = {
							OR = {
								province_id = 118
								province_id = 146
								province_id = 358
								province_id = 148
								province_id = 2296
							}
							is_core = ROOT
							OR = {
								has_building = temple
								has_building = cathedral
							}
						}
						amount = 5
					}
				}
			}
		}
		effect = {
			change_religion = hellenic_pagan
			capital_scope = {
				change_religion = hellenic_pagan
				add_permanent_province_modifier = {
					name = "religious_zeal_at_conv"
					duration = 4500
				}
			}
			every_owned_province = {
				limit = {
					OR = {
						province_id = 118
						province_id = 146
						province_id = 358
						province_id = 148
						province_id = 2296
					}
					OR = {
						has_building = temple
						has_building = cathedral
					}
					NOT = { is_capital = yes } # Need to test to see if this exclude the "bug" on the ingame textbox, whereas if your capital is also a holy site it gets listed and changed twice.
				}
				change_religion = hellenic_pagan
				add_permanent_province_modifier = {
					name = "religious_zeal_at_conv"
					duration = 4500
				}
			}
			if = {
				limit = {
					OR = {
						tag = ROM # Roman Empire
						tag = BYZ # Byzantine Empire
						tag = GRE # Greece
						culture_group = byzantine
						culture_group = latin
					}
				}
				add_stability = -2
			}
			if = {
				limit = {
					NOT = {
						tag = ROM # Roman Empire
						tag = BYZ # Byzantine Empire
						tag = GRE # Greece
						culture_group = byzantine
						culture_group = latin
					}
				}
				add_stability = -4
			}
			add_country_modifier = {
				name = holy_site_conversion_zeal
				duration = 3650
			}
			set_country_flag = ctr_hellenic_decision
			hidden_effect = {
				add_country_modifier = {
					name = has_recently_converted_ctr_timer
					duration = 3650
					hidden = yes
				}
			}
		}
		ai_will_do = {
			factor = 0
		}
	}
